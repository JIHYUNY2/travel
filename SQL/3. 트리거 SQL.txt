--LIST_TRAVEL과 1:1 상태라 LIST_TRAVEL에 새로운 데이터가 추가 되면 자동으로 TRAVEL_SEASON에 추가됨 
CREATE OR REPLACE TRIGGER TRG_TRAVEL_SEASON
    AFTER INSERT ON LIST_TRAVEL
    FOR EACH ROW
    BEGIN
        INSERT INTO TRAVEL_SEASON VALUES (:NEW.TRAVEL_NO, DEFAULT, DEFAULT, DEFAULT, DEFAULT);
    END TRG_TRAVEL_SEASON;
